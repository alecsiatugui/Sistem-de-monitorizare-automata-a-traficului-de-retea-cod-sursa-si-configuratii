R1 – CENTRAL ROUTER
enable
conf t
hostname R1
interface GigabitEthernet1
 description TO-R2
 ip address 10.0.1.1 255.255.255.252
 no shutdown
interface GigabitEthernet2
 description TO-R3
 ip address 10.0.2.1 255.255.255.252
 no shutdown
ip route 0.0.0.0 0.0.0.0 10.0.0.1
ip route 172.16.100.0 255.255.255.0 10.0.2.2
ip route 192.168.10.0 255.255.255.0 10.0.1.2
ip route 192.168.20.0 255.255.255.0 10.0.1.2
ip route 192.168.30.0 255.255.255.0 10.0.1.2
username cisco privilege 15 secret cisco
ip domain-name lab.local
crypto key generate rsa modulus 2048
ip ssh version 2
netconf-yang
copy running-config startup-config

SWITCH-L2-1 – LAN USER ACCESS SWITCH
enable
conf t
interface e0/0
 no shutdown
 switchport mode access
interface e0/1
 no shutdown
 switchport mode access
 spanning-tree portfast
interface e0/2
 no shutdown
 switchport mode access
 spanning-tree portfast
end
write memory

SWITCH-L2-2 – VLAN ACCESS SWITCH
enable
conf t
vlan 20
 name USERS_VLAN20
vlan 30
 name USERS_VLAN30
interface e0/0
 switchport trunk encapsulation dot1q
 switchport mode trunk
 no shutdown
interface e0/1
 switchport mode access
 switchport access vlan 20
 no shutdown
interface e0/2
 switchport mode access
 switchport access vlan 30
 no shutdown
vlan 99
 name MANAGEMENT
interface vlan 99
 ip address 192.168.99.2 255.255.255.0
 no shutdown
ip default-gateway 192.168.99.1
end
write memory
